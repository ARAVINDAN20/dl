                    from tensorflow.keras import layers
                    import numpy as np (*)

                        [0, 0],
                        [0, 1],
                        [1, 0],

# Demonstrating basic data types
                    import pandas as pd
                    import tensorflow as tf
                    from tensorflow.keras import layers
                    
                    # Load dataset
                    data = pd.read_csv('/content/Housing.csv')
                    
                    # Convert categorical columns
                    data = pd.get_dummies(data, drop_first=True)
                    
                    # Make sure all data is numeric
                    x = data.drop('price', axis=1).astype('float32')
                    y = data['price'].astype('float32')
                    
                    # Build model
                    model = tf.keras.Sequential([
                        tf.keras.Input(shape=(x.shape[1],)),
                        layers.Dense(64, activation='relu'),
                        layers.Dense(1)
                    ])
                    
                    # Compile model (add both MSE and MAE as metrics)
                    model.compile(optimizer='adam',
                                  loss='mse',
                                  metrics=['mse', 'mae'])
                    
                    # Train model
                    model.fit(x, y, epochs=50)
                    
                    # Evaluate model
                    loss, mse, mae = model.evaluate(x, y)
                    print("MSE:", mse)
                    print("MAE:", mae)

Lab 5: timeseries

▶

A simple introduction to Python with a basic Hello World program.


            #Time Series Prediction Using Simple RNN
import numpy as np
import matplotlib.pyplot as plt
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import SimpleRNN, Dense
from sklearn.preprocessing import MinMaxScaler

def generate_data(n_samples=100, timesteps=10):
    x = np.linspace(0, 50, n_samples)
    y = np.sin(x) + np.random.normal(0, 0.1, n_samples)
    return y

data = generate_data()
scaler = MinMaxScaler(feature_range=(0, 1))
data_scaled = scaler.fit_transform(data.reshape(-1, 1))

X, y = [], []
seq_length = 10
for i in range(len(data_scaled) - seq_length):
    X.append(data_scaled[i:i + seq_length])
    y.append(data_scaled[i + seq_length])
X, y = np.array(X), np.array(y)

split = int(len(X) * 0.8)
X_train, X_test = X[:split], X[split:]
y_train, y_test = y[:split], y[split:]

model = Sequential([
    SimpleRNN(20, input_shape=(seq_length, 1)),
    Dense(1)
])

model.compile(optimizer='adam', loss='mean_squared_error')

model.fit(X_train, y_train, epochs=5, batch_size=8, validation_data=(X_test, y_test))

y_pred = model.predict(X_test)
y_pred = scaler.inverse_transform(y_pred)
y_test = scaler.inverse_transform(y_test.reshape(-1, 1))

plt.plot(y_test, label="Actual Values")
plt.plot(y_pred, label="Predicted Values")
plt.legend()
plt.xlabel("Time")
plt.ylabel("Value")
plt.title("Simple RNN for Time Series Prediction")
plt.show()

import tensorflow as tf
import numpy as np

# Directly open the local file (no get_file needed)
path = '/content/ss.txt'
text = open(path, 'rb').read().decode(encoding='utf-8')

# Create character to index and index to character mappings
chars = sorted(set(text))
char2idx = {c: i for i, c in enumerate(chars)}
idx2char = np.array(chars)

# Prepare the dataset
seq_length = 100
x = []
y = []
for i in range(len(text) - seq_length):
    x.append([char2idx[c] for c in text[i:i + seq_length]])
    y.append(char2idx[text[i + seq_length]])

x = np.array(x)
y = np.array(y)

# Build the model
model = tf.keras.Sequential([
    tf.keras.Input(shape=(seq_length,)),
    tf.keras.layers.Embedding(len(chars), 256),
    tf.keras.layers.LSTM(256),
    tf.keras.layers.Dense(len(chars), activation='softmax')
])

# Compile the model
model.compile(optimizer='adam', loss='sparse_categorical_crossentropy', metrics=['accuracy'])

# Train the model
model.fit(x, y, epochs=1)
loss, acc = model.evaluate(x, y)
print("Test Accuracy:", acc)

Lab 8: encoderdecoder

▶

A simple introduction to Python with a basic Hello World program.


            import tensorflow as tf
import numpy as np
import matplotlib.pyplot as plt

# Data
input_texts = ['hello', 'hi', 'thanks', 'goodbye']
target_texts = ['salut', 'salut', 'merci', 'au revoir']

input_chars = sorted(set(''.join(input_texts)))
target_chars = sorted(set(''.join(target_texts)))

input_index = {c:i for i,c in enumerate(input_chars)}
target_index = {c:i for i,c in enumerate(target_chars)}
reverse_target_index = {i:c for c,i in target_index.items()}

encoder_input = np.zeros((len(input_texts), max(len(x) for x in input_texts), len(input_chars)))
decoder_input = np.zeros((len(input_texts), max(len(x) for x in target_texts), len(target_chars)))
decoder_target = np.zeros_like(decoder_input)

for i, (inp, tar) in enumerate(zip(input_texts, target_texts)):
    for t, char in enumerate(inp):
        encoder_input[i, t, input_index[char]] = 1
    for t, char in enumerate(tar):
        decoder_input[i, t, target_index[char]] = 1
        if t > 0:
            decoder_target[i, t-1, target_index[char]] = 1

# Encoder
enc_inp = tf.keras.Input(shape=(None, len(input_chars)))
enc_out, state_h, state_c = tf.keras.layers.LSTM(128, return_state=True)(enc_inp)
enc_states = [state_h, state_c]

# Decoder
dec_inp = tf.keras.Input(shape=(None, len(target_chars)))
dec_lstm = tf.keras.layers.LSTM(128, return_sequences=True, return_state=True)
dec_out, _, _ = dec_lstm(dec_inp, initial_state=enc_states)
dec_dense = tf.keras.layers.Dense(len(target_chars), activation='softmax')
dec_out = dec_dense(dec_out)

# Model
model = tf.keras.Model([enc_inp, dec_inp], dec_out)
model.compile(optimizer='adam', loss='categorical_crossentropy')
model.fit([encoder_input, decoder_input], decoder_target, epochs=100, verbose=0)

# Encoder Model (for prediction)
encoder_model = tf.keras.Model(enc_inp, enc_states)

# Decoder Model (for prediction)
dec_state_input_h = tf.keras.Input(shape=(128,))
dec_state_input_c = tf.keras.Input(shape=(128,))
dec_states_inputs = [dec_state_input_h, dec_state_input_c]
dec_out2, state_h2, state_c2 = dec_lstm(dec_inp, initial_state=dec_states_inputs)
dec_states2 = [state_h2, state_c2]
dec_out2 = dec_dense(dec_out2)
decoder_model = tf.keras.Model([dec_inp] + dec_states_inputs, [dec_out2] + dec_states2)

# Small decode function
def decode_sequence(x):
    s = encoder_model.predict(x)
    t = np.zeros((1,1,len(target_chars)))
    o = ''
    for _ in range(20):
        y,h,c = decoder_model.predict([t]+s)
        i = np.argmax(y[0,-1,:])
        o += reverse_target_index[i]
        if o[-1]=='\n': break
        t = np.zeros((1,1,len(target_chars)))
        t[0,0,i]=1
        s=[h,c]
    return o

# Encode "hello" and translate
input_seq = np.zeros((1, max(map(len, input_texts)), len(input_chars)))
for t,c in enumerate('hello'): input_seq[0,t,input_index[c]]=1

print('Input: hello\nPredicted:', decode_sequence(input_seq))
